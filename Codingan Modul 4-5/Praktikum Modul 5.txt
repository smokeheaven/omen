Interface Sensor Prox

#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd (0x27,16,2);
int ProxSensor = A1; 
int Buzzer = 9; 
int button = 8;
int led1= 10;
int inputVal = 0;
int count, cout;
void setup() {
 pinMode(Buzzer, OUTPUT);
 pinMode(ProxSensor, INPUT);
 pinMode(led1, OUTPUT);
 Serial.begin(9600);
 lcd.init();
 lcd.backlight();
}

void loop() {
if(digitalRead(button) == LOW){
  count++;
  cout++;
}
  if(count >= 1){
      if (digitalRead(ProxSensor) == LOW) {
      Serial.println("Terdeteksi Halangan!");
      tone(Buzzer, 123);
      inputVal = analogRead(ProxSensor);
      Serial.println(inputVal);
      delay(1000);
      if(cout >= 2){
        lcd.setCursor(2,0);
        lcd.print("HALANGAN");
        delay(100);
        if(cout >=3){
          digitalWrite(led1, HIGH);
          delay(1000);
          if(cout >=3){
            cout =1;
          }
        }
      }
      } 
      
      if (digitalRead(ProxSensor) == HIGH) {
      Serial.println("Clear!");
      tone(Buzzer, 65);
      inputVal = analogRead(ProxSensor);
      Serial.println(inputVal);
      delay(1000);
       if(cout >= 2){
        lcd.setCursor(2,0);
        lcd.print("AMAN AZA");
        delay(100);
        if(cout >=3){
          digitalWrite(led1, LOW);
          delay(1000);
          if(cout >=3){
            cout =1;
          }
        }
      }
      } 
  }
}
